boolean[] Barr = new boolean[3];
void setup() {
  getRandom ();
  print(Barr[0] +" "+ Barr[1]+ " "+ Barr[2]);
}

void getRandom () {
  float r = random(0, 4);
  println(r);
  if (r>0 && r<1) {
    Barr[0] = true;
    Barr[1] = true;
    Barr[2] = false;
  } else if (r>1 && r<2) {
    Barr[0] = true;
    Barr[1] = false;
    Barr[2] = true;
  } else if (r>2 && r<4) {
    Barr[0] = false;
    Barr[1] = true;
    Barr[2] = true;
  }
}
